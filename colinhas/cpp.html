                return 0;
            }

#include <bits/stdc++.h>
using namespace std;

string filter(const string& texto, const string& filtro){
    string aux;
    copy_if(texto.begin(), texto.end(), back_inserter(aux), [&filtro](char c){
        return filtro.find(c) != string::npos;
    });
    return aux;
}

int main(){
    string filtro;
    string texto;

    cout << "Digite o filtro a ser aplicado:\n";
    getline(cin, filtro);
    cout << "Digite o texto a receber o filtro:\n";
    getline(cin, texto);

    cout << filter(texto, filtro) << endl;
    return 0;
}

Histograma

#include <bits/stdc++.h>
#include <map>
using namespace std;

/* Objetivo: Retornar a quantidade de incidências de cada caractere em uma string */
map<char,int> histogram(const string& s){
    map<char,int> h;
    for (auto c : s) {
        ++h[c];
    }
    return h;
}

int main(){
    string txt = "ola mundo";
    auto mp = histogram(txt);
    for(auto x : mp){
        cout << x.first << " - " << x.second << endl;
    }
    return 0;
}

#include <bits/stdc++.h>

string smap(const string& strOriginal, const function<char(char)>& f){
    string strFormatada;
    for (auto caracter : strOriginal){
        strFormatada.push_back(f(caracter));

    return strFormatada;

Regex

#include <bits/stdc++.h>
using namespace std;
int main(){
    string texto;
    cin >> texto;
    
    // Exemplo: 123.456.789-10
    if(regex_match(texto, regex("([0-9]{3}[.]){2}[0-9]{3}[-][0-9]{2}"))){
        cout << "OK!" << endl;
    } else {
        cout << "ERRADO" << endl;
    }
    return 0;
}

/*
Notas:
[az] -> a ou z 
[a-z] -> de a até z
[a][z] -> a e z juntos, ex: "az" 
[a]{q} -> a ocorre q vezes, ex: [b]{3} para "bbb"
[a]+ -> a ocorre 1 ou mais vezes, ex: "b", "bb", ...
[a]* -> a ocorre 0 ou mais vezes, ex: "", "b", "bb", ...
[a-z]{q} -> q elementos do intervalo, ex: [a-f]{3} para "afd"
[a-z]+ -> 1 ou mais elementos do intervalo
[a-z]* -> 0 ou mais elementos do intervalo

$ -> fim da linha
^ -> início da linha
^[ab]$ -> busca exatamente por "ab", similar ao regex_match que retorna true somente quando é igual.
() -> agrupa padrões
*/

Token

#include <bits/stdc++.h>
using namespace std;

vector<string> tokenizar(const string& texto, const char* delimitador) {
    vector<string> splitado;

    // Criar uma cópia mutável da string original
    char c_texto[texto.size() + 1];
    strcpy(c_texto, texto.c_str());

    // Usar strtok para dividir a string
    char* token = strtok(c_texto, delimitador);
    while (token != nullptr) {
        splitado.push_back(token);
        token = strtok(nullptr, delimitador);
    }

    return splitado;
}

int main() {
    string texto;
    getline(cin, texto);

    vector<string> split = tokenizar(texto, " ");

    for (int i = 0; i < (int)split.size(); i++) {
        cout << split[i] << endl;
    }

    return 0;
}

Algoritmos de Busca

Busca completa - quantas vezes uma string A ocorre em B

          
            #include <iostream>
              #include <string>
              
              using namespace std;
              
              int occurrences(const string& P, const string& S) {
                  int m = P.size();
                  int n = S.size();
                  int occ = 0;
              
                  for (int i = 0; i < n - m + 1; ++i)
                      occ += (P == S.substr(i, m) ? 1 : 0);
              
                  return occ;
              }
              
              int main() {
                  string P = "ab";
                  string S = "ababab";
                  cout << "Número de ocorrências: " << occurrences(P, S) << endl; //3
                  return 0;
              }
